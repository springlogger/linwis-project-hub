Priority Badges

Vue component — PriorityBadge.vue
<!-- PriorityBadge.vue -->
<template>
  <span :class="badgeClass">{{ priority }}</span>
</template>

<script setup lang="ts">
const props = defineProps<{ priority: 'High' | 'Medium' | 'Low' }>()

const badgeClass = computed(() => {
  const base = 'inline-flex items-center px-1.5 py-0.5 rounded text-2xs font-semibold tracking-wide capitalize'
  return {
    High:   base + ' bg-status-danger/15 text-status-danger',
    Medium: base + ' bg-status-warning/15 text-status-warning',
    Low:    base + ' bg-status-success/15 text-status-success',
  }[props.priority]
})
</script>

@layer base {
  :root {
    /* Used for non-Tailwind inline contexts */
    --color-base:      theme('colors.surface.base');
    --color-elevated:  theme('colors.surface.elevated');
    --color-overlay:   theme('colors.surface.overlay');
    --color-primary:   theme('colors.brand.primary');
    --color-accent:    theme('colors.brand.accent');
  }

  html { @apply bg-surface-base text-content-primary font-sans antialiased; }
  ::selection { @apply bg-brand-primary/30 text-content-primary; }

  ::-webkit-scrollbar       { @apply w-1 h-1; }
  ::-webkit-scrollbar-track { background: transparent; }
  ::-webkit-scrollbar-thumb { @apply bg-surface-overlay rounded-full; }
